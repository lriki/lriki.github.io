225 inline bool notifyException(const char* file, int line, const char* exprString, TArgs... args)
226 {
227  const size_t BUFFER_SIZE = 512;
228  Char str[BUFFER_SIZE] = {};
229  errorPrintf(str, BUFFER_SIZE, args...);
230  TException e(str);
231  detail::Exception_setSourceLocationInfo(e, file, line, exprString);
232  auto h = Exception::notificationHandler();
233  if (h != nullptr && h(e)) return true;
234  printError(e);
235  throw e;
236  return true;
237 }
